insta_box = 
  instacart %>% 
  filter(!is.na(department)) %>% 
  mutate(department = fct_reorder(department, order_hour_of_day)) %>% 
  ggplot(aes(x = department, y = order_hour_of_day, fill = department)) +
  geom_boxplot() +
  theme(axis.text.x = element_text(angle = 90, hjust = 1))
